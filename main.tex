//Distance Formula
double Point::operator -(const Point& p2) const
{
  double t1, t2, d;
  t1 = p2.xCoord - xCoord; //(x2-x1)
  t2 = p2.yCoord - yCoord; //(y2-y1)
  d = sqrt( pow(t1,2) + pow(t2,2) );
  return d;
}
